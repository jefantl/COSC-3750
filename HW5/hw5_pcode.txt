if no arguments passed to program:
    read from stdin
    write to stdout
else:
    for each argument in passed arguments:
        if argument is '-':
            read from stdin
            write to stdout 
        else:
            open file
            if file is valid:
                read from file
                write to stdout
            else:
                report this is not a valid file
